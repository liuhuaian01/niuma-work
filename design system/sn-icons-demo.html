Super Niuma Icon System

创作脉冲 — Pure geometric stroke icons. 16 icons, 24×24 viewBox, monochrome.

Usage

<!-- Inline the sprite at the top of your page -->
<svg style="display:none">
  <!-- ... symbols ... -->
</svg>

<!-- Use anywhere with currentColor inheritance -->
<span class="sn-icon">
  <svg><use href="#icon-search"/></svg>
</span>

<!-- Size variants -->
<span class="sn-icon sn-icon--sm">...</span>
<span class="sn-icon sn-icon--lg">...</span>

<!-- Color via class (or set color directly) -->
<span class="sn-icon sn-icon--blue">...</span>

<!-- Interactive icon button -->
<button class="sn-icon-btn">
  <span class="sn-icon"><svg><use href="#icon-close"/></svg></span>
</button>
